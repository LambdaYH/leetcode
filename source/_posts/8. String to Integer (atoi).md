#### [8. String to Integer (atoi)](https://leetcode.com/problems/string-to-integer-atoi/)
##### 第一次
```c++
class Solution
{
public:
    int myAtoi(string s)
    {
        int sum = 0;
        int positive = 1;
        bool isEntered = false;
        for (const auto &ch : s)
        {
            switch (ch)
            {
            case '+':
                if (!isEntered)
                {
                    positive = 1;
                    isEntered = true;                    
                }
                else
                    return positive * sum;
                break;
            case '-':
                if (!isEntered)
                {
                    positive = -1;
                    isEntered = true;                    
                }
                else
                    return positive * sum;
                break;
            default:
                if (ch >= 48 && ch <= 57)
                {
                    if (!isEntered)
                        isEntered = true;
                    int pop = ch - 48;
                    switch (positive)
                    {
                    case 1:
                        if (sum > INT_MAX / 10 || (sum == INT_MAX / 10 && pop > 7))
                            return INT_MAX;
                        break;
                    case -1:
                        if (-sum < INT_MIN / 10 || (-sum == INT_MIN / 10 && pop >= 8))
                            return INT_MIN;
                        break;
                    }
                    sum = sum * 10 + pop;
                }
                else if (ch != ' ')
                {
                    return positive * sum;
                }else if (ch == ' ' && isEntered)
                {
                    return positive * sum;
                }
            }
        }
        return positive * sum;
    }
};
```
